Ejercicio 5

var numero1 = 5;
var numero2 = 8;

if(...) {
alert("numero1 no es mayor que numero2");
}
if(...) {
alert("numero2 es positivo");
}
if(...) {
alert("numero1 es negativo o distinto de cero");
}
if(...) {
alert("Incrementar en 1 unidad el valor de
numero1 no lo hace mayor o igual que numero2");
}
